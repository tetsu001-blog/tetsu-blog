{{- $url := .Get 0 -}} {{- $videoID := "" -}} {{- if findRE "youtu\\.be/" $url -}} {{- $videoID = replaceRE ".*youtu\\.be/([^?&/]+).*" "$1" $url -}} {{- else if findRE "[?&]v=" $url -}} {{- $videoID = replaceRE ".*[?&]v=([^&]+).*" "$1" $url -}} {{- else if findRE "/embed/" $url -}} {{- $videoID = replaceRE ".*/embed/([^?&/]+).*" "$1" $url -}} {{- else -}} {{- $videoID = $url -}} {{- end -}}
